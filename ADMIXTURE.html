sample_ids <- samples$V2   # V2 = individual IDs


readQ <- function(K) {
  qfile <- paste0(qpath, "md6ct95ms200_numeric.", K, ".Q")
  q <- read.table(qfile, stringsAsFactors = FALSE)
  colnames(q) <- paste0("Pop", 1:K)
  q$SampleID <- sample_ids
  q$K <- K
  q }
